# ✅ 5. Reward Logic (CRITICAL)

Trigger when user becomes full provider:

```ts id="reward_logic"
function handleProviderCompletion(user: User) {
  if (!user.referredBy) return;

  const referral = referrals.find(
    r => r.referredUserId === user.id && !r.rewarded
  );

  if (!referral) return;

  const referrer = users.find(
    u => u.referralCode === user.referredBy
  );

  if (!referrer) return;

  // reward
  referrer.tangaBalance = (referrer.tangaBalance || 0) + 3;

  // log transaction
  addTangaTransaction({
    userId: referrer.id,
    amount: 3,
    type: "referral_reward",
    description: "Referral bonus",
  });

  // update referral
  referral.status = "completed";
  referral.rewarded = true;

  saveAll();
}
```
