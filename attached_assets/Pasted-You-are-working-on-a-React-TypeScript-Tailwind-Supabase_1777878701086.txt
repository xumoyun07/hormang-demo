# ✅ 3. Eligibility Logic

Create helper:

```ts
function getLast10RejectedOffers(providerId: string) {
  const providerOffers = offers
    .filter(o => o.providerId === providerId)
    .sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime())
    .slice(0, 10);

  const eligible =
    providerOffers.length === 10 &&
    providerOffers.every(o => o.status === "rejected" && !o.refunded);

  return { providerOffers, eligible };
}
```

---

# ✅ 4. Refund Calculation

```ts
function calculateRefund(offers: Offer[]) {
  const totalSpent = offers.reduce((sum, o) => sum + o.tangaSpent, 0);
  return Math.floor(totalSpent * 0.5);
}
```

---

# ✅ 5. Refund Action (Admin Trigger)

```ts
function refundProvider(providerId: string, offers: Offer[]) {
  const refundAmount = calculateRefund(offers);

  // update user balance
  provider.tangaBalance += refundAmount;

  // create transaction
  addTangaTransaction({
    userId: providerId,
    amount: refundAmount,
    type: "admin_adjustment",
    source: "refund_last_10_rejected",
    description: "50% refund for last 10 rejected offers",
  });

  // mark offers
  offers.forEach(o => (o.refunded = true));

  saveAll();
}
```
