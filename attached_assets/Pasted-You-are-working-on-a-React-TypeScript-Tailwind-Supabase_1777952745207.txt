# ✅ 1. Replace Image URL Input with File Upload

## Remove:

```tsx
<input value={f.image} ... />
```

---

## Add:

```tsx
<input
  type="file"
  accept="image/*"
  onChange={handleImageUpload}
/>
```

---

## Implement handler:

```ts
function handleImageUpload(e: React.ChangeEvent<HTMLInputElement>) {
  const file = e.target.files?.[0];
  if (!file) return;

  const reader = new FileReader();
  reader.onloadend = () => {
    set("image", reader.result as string); // base64
  };
  reader.readAsDataURL(file);
}
```
